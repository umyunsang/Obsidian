### MyController 

```java
@Controller
public class MyController{
	@GetMapping("/")
	public String home(){
		return "home";
	}
	
	@GetMapping("/ex01")
	public String ex01(){
		return "ex01";
	}
	
	@GetMapping("/ex01/answer")
	public String ex01Answer(@RequestParam(name="mid") String mid, Model mo){
		mo.addAttribute("mid",mid);
		return "ex01Answer";
	}
	
	@GetMapping("/ex02")
	public String ex02(){
		return "ex02";
	}
	
	@GetMapping("/ex02/answer")
	public String ex02Answer(@RequestParam(name="mname") String mname,@RequestParam(name="po") String po, Model mo){
		mo.addAttribute("mname",mname);
		mo.addAttribute("po",po);
		int salary = 0;
		switch(po) {
			case "팀장" -> salary = 7000;
		}
		mo.addAttribute("salary", salary);
		return "ex02Answer";
	}
	@GetMapping("/ex03")
		public String ex03(){
			return "ex03";
	}
	@GetMapping("/ex03/answer")
		public String ex03Answer(@RequestParam(name="mname") String mname, @RequestParam(name="color") String color, Model mo) {
		mo.addAttribute("mname", mname);
		mo.addAttribute("color", color);
		return "ex03Answer"
	}
	@GetMapping("/ex04")
	public String ex04(Model mo){
		var arr = new ArrayList<String>();
		arr.add("고흐");
		arr.add("james");
		arr.add("dooli");
		arr.add("bread");
		mo.addAttribute("arr",arr);
		return "ex04";
	}

	@GetMapping("/login")
	public String login(){
		return "login";
	}
	@GetMapping("/login/check")
	public String logincheck(HttpsSession se,RedirectArributes re, @RequestParam("mid") String mid){
		var arr = new ArrayList<String>();
		arr.add("고흐");
		arr.add("james");
		arr.add("dooli");
		arr.add("bread");
		if(arr.contrain(mid)){
			se.setAttribute("mid", mid);
			return "redirect:/menu";
		}
		else {
			re.addAttribute("msg", mid +"는 미등록 아이디");
			return "redirect:/popup"
		}
		
	}
	@GetMapping("/popup")
	public String popup(@RequestParam("msg") String msg, Model mo){
		mo.addAttribute("msg", msg);
		return "popup"
	}
	@GetMapping("/menu")
	public String menu(HttpsSession se, Model mo){
		mo.addAttribute("mid", se.getAttribute("mid"));
		return "menu"
	}
	@GetMapping("/coffee")
	public String coffee(HttpsSession se, Model mo,){
		mo.addAttribute("mid",se.getAttribute("mid"));
		return "coffee";
	}
	@GetMapping("/logout")
	public String logout(HttpsSession se, Model mo){
		mo.addAttribute("mid", se.getAttribute("mid"));
		se.invalidate();
		return "logout"
	}
}
```
