### 원형 연결 리스트(Circular Linked List)

파이썬으로 원형 연결 리스트를 구현하는 예시 코드입니다.

```python
class Node:
    def __init__(self, data):
        self.data = data
        self.next = None

class CircularLinkedList:
    def __init__(self):
        self.head = None

    # 리스트 끝에 노드를 추가하는 메서드
    def append(self, data):
        new_node = Node(data)
        if not self.head:
            self.head = new_node
            self.head.next = self.head
        else:
            current = self.head
            while current.next != self.head:
                current = current.next
            current.next = new_node
            new_node.next = self.head

    # 리스트를 순회하며 노드들의 데이터를 출력하는 메서드
    def display(self):
        current = self.head
        while current:
            print(current.data, end=" -> ")
            current = current.next
            if current == self.head:
                break
        print("HEAD")

    # 데이터를 수정하는 메서드
    def modify(self, old_data, new_data):
        current = self.head
        while current:
            if current.data == old_data:
                current.data = new_data
                return
            current = current.next
            if current == self.head:
                break
        print("Data not found")

    # 데이터를 삭제하는 메서드
    def delete(self, data):
        if not self.head:
            print("List is empty")
            return

        # 리스트에 노드가 하나뿐인 경우
        if self.head.data == data and self.head.next == self.head:
            self.head = None
            return

        prev = None
        current = self.head
        while current:
            if current.data == data:
                if current == self.head:  # 첫 번째 노드 삭제 시
                    while current.next != self.head:
                        current = current.next
                    current.next = self.head.next
                    self.head = self.head.next
                else:
                    prev.next = current.next
                return
            prev = current
            current = current.next
            if current == self.head:
                break
        print("Data not found")

    # 데이터를 찾는 메서드
    def search(self, data):
        index = 0
        current = self.head
        while current:
            if current.data == data:
                print(f"Data {data} found at index {index}")
                return
            current = current.next
            index += 1
            if current == self.head:
                break
        print("Data not found")

    # 특정 데이터 뒤에 데이터를 추가하는 메서드
    def insert_after(self, target_data, new_data):
        new_node = Node(new_data)
        current = self.head
        while current:
            if current.data == target_data:
                new_node.next = current.next
                current.next = new_node
                return
            current = current.next
            if current == self.head:
                break
        print("Target data not found")
```
