### 원형 연결 리스트(Circular Linked List)

```python
class Node:
    def __init__(self, data):
        self.data = data
        self.next = None

class CircularLinkedList:
    def __init__(self):
        self.head = None

    # 1. HEAD 생성
    def create_head(self, data):
        new_node = Node(data)
        new_node.next = new_node  # 자기 자신을 가리키는 원형
        self.head = new_node

    # 2. 마지막 위치에 데이터 추가
    def append(self, data):
        # 새로운 노드를 생성합니다.
        new_node = Node(data)
        # 리스트가 비어있으면 새로운 노드를 HEAD로 지정합니다.
        if not self.head:
            self.create_head(data)
            return

        # 리스트의 마지막 노드를 찾습니다.
        current = self.head
        while current.next != self.head:
            current = current.next
        # 마지막 노드의 다음에 새로운 노드를 연결하고,
        current.next = new_node
        # 새로운 노드의 다음을 HEAD로 지정하여 원형으로 연결합니다.
        new_node.next = self.head

    # 3. 데이터 삽입
    def insert(self, data):
        # 새로운 노드를 생성합니다.
        new_node = Node(data)
        if not self.head:  
            # 리스트가 비어있으면 새로운 노드를 HEAD로 지정합니다.
            new_node.next = new_node  
            self.head = new_node  
        else:
            # 리스트의 마지막 노드를 찾습니다.
            temp = self.head
            while temp.next != self.head:
                temp = temp.next
            # 마지막 노드의 다음에 새로운 노드를 연결하고,
            temp.next = new_node  
            # 새로운 노드의 다음을 HEAD로 지정하여 원형으로 연결합니다.
            new_node.next = self.head  

            
    # 4. 데이터 전체 순차 출력
    def display(self):
        # 리스트가 비어있는지 확인합니다.
        if not self.head:
            print("리스트가 비어있습니다.")
            return

        # current 변수에 HEAD를 할당합니다.
        current = self.head
        while True:
         # 현재 노드의 데이터를 출력합니다.
            print(current.data, end=" -> ")
         # 다음 노드로 이동합니다.
            current = current.next
         # 다음 노드가 HEAD일 경우, 리스트의 처음으로 되돌아왔으므로 반복을 종료합니다.
            if current == self.head:
                break
        # HEAD를 나타내는 마지막 노드를 출력합니다.
        print("HEAD")


    # 5. 데이터 삭제
    def delete(self, data):
		# 리스트가 비어있는지 확인합니다.
        if not self.head:
            print("리스트가 비어있습니다.")
            return

        # 삭제할 데이터가 HEAD에 있는 경우
        if self.head.data == data:
            # 리스트에 노드가 하나뿐인 경우
            if self.head.next == self.head:  
                self.head = None
            else:
            
 # HEAD를 마지막 노드로 이동시키고, 다음 노드를 HEAD로 지정하여 첫 번째 노드를 삭제합니다.
                current = self.head
                while current.next != self.head:
                    current = current.next
                current.next = self.head.next
                self.head = self.head.next
            return

 # 삭제할 데이터가 HEAD가 아닌 경우
        prev = None
        current = self.head
        while current:
            if current.data == data:
            
 # 이전 노드의 다음 노드를 삭제 대상 노드의 다음 노드로 설정하여 삭제합니다.
                prev.next = current.next
                return
            prev = current
            current = current.next
            if current == self.head:
                break
        print("Data not found")


    # 6. 데이터 수정
    def modify(self, old_data, new_data):
        # 리스트가 비어있는지 확인합니다.
        if not self.head:
            print("리스트가 비어있습니다.")
            return

        # 리스트를 순회하며 수정할 데이터를 찾습니다.
        current = self.head
        while True:
            # 현재 노드의 데이터가 수정 전 데이터와 일치하는 경우
            if current.data == old_data:
                # 데이터를 새로운 데이터로 변경합니다.
                current.data = new_data
                return
            current = current.next
            # HEAD에 도달하면 반복을 종료합니다.
            if current == self.head:
                break
        # 수정 전 데이터가 리스트에 없는 경우
        print("Data not found")


    # 7. 원하는 위치 검색
    def search(self, data):
        # 리스트가 비어있는지 확인합니다.
        if not self.head:
            print("리스트가 비어있습니다.")
            return

        # 리스트를 순회하며 검색할 데이터를 찾습니다.
        current = self.head
        index = 0
        while True:
            # 현재 노드의 데이터가 검색할 데이터와 일치하는 경우
            if current.data == data:
                # 해당 데이터와 인덱스를 출력합니다.
                print(f"Data {data} found at index {index}")
                return
            current = current.next
            index += 1
            # HEAD에 도달하면 반복을 종료합니다.
            if current == self.head:
                break
        # 검색한 데이터가 리스트에 없는 경우
        print("Data not found")
```
