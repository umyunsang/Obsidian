### 원형 연결 리스트(Circular Linked List)

```python
class Node:
    def __init__(self, data):
        self.data = data
        self.next = None

class CircularLinkedList:
    def __init__(self):
        self.head = None

    # 1. HEAD 생성
    def create_head(self, data):
        new_node = Node(data)
        new_node.next = new_node  # 자기 자신을 가리키는 원형
        self.head = new_node

    # 2. 마지막 위치에 데이터 추가
    def append(self, data):
        new_node = Node(data)
        if not self.head:
            self.create_head(data)
            return

        current = self.head
        while current.next != self.head:
            current = current.next
        current.next = new_node
        new_node.next = self.head

    # 3. 데이터 삽입
    def insert(self, data):
        new_node = Node(data)
        if not self.head:  
            new_node.next = new_node  
            self.head = new_node  
        else:
            temp = self.head
            while temp.next != self.head:
                temp = temp.next
            temp.next = new_node  
            new_node.next = self.head  
            
    # 4. 데이터 전체 순차 출력
    def display(self):
        if not self.head:
            print("리스트가 비어있습니다.")
            return

        current = self.head
        while True:
            print(current.data, end=" -> ")
            current = current.next
            if current == self.head:
                break
        print("HEAD")

    # 5. 데이터 삭제
    def delete(self, data):
        if not self.head:
            print("리스트가 비어있습니다.")
            return

        if self.head.data == data:
            if self.head.next == self.head:  # 리스트에 노드가 하나뿐인 경우
                self.head = None
            else:
                current = self.head
                while current.next != self.head:
                    current = current.next
                current.next = self.head.next
                self.head = self.head.next
            return

        prev = None
        current = self.head
        while current:
            if current.data == data:
                prev.next = current.next
                return
            prev = current
            current = current.next
            if current == self.head:
                break
        print("Data not found")

    # 6. 데이터 수정
    def modify(self, old_data, new_data):
        if not self.head:
            print("리스트가 비어있습니다.")
            return

        current = self.head
        while True:
            if current.data == old_data:
                current.data = new_data
                return
            current = current.next
            if current == self.head:
                break
        print("Data not found")

    # 7. 원하는 위치 검색
    def search(self, data):
        if not self.head:
            print("리스트가 비어있습니다.")
            return

        current = self.head
        index = 0
        while True:
            if current.data == data:
                print(f"Data {data} found at index {index}")
                return
            current = current.next
            index += 1
            if current == self.head:
                break
        print("Data not found")
```
